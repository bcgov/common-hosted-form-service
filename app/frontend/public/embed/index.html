CHEFS Form Viewer Widget

A web component for embedding CHEFS forms in external applications

Note: This is a demo page. Replace the form-id and api-key with your actual values to see the widget in action.

Quick Start

Include the web component script and use it in your HTML:

<!-- Include the web component -->
 <script src="/widgets/chefs-form-viewer.js"></script>

<!-- Use the component -->
<chefs-form-viewer
  form-id="your-form-id"
  api-key="your-api-key"
  theme="bcgov">
</chefs-form-viewer>

Event Handling

Listen for form events to integrate with your application:

// Form submitted successfully
document.addEventListener('formSubmit', (event) => {
  console.log('Form submitted:', event.detail.submission);
  // Handle successful submission
});

// Form submission error
document.addEventListener('formError', (event) => {
  console.error('Form error:', event.detail.error);
  // Handle error
});

// Form data changed
document.addEventListener('formChange', (event) => {
  console.log('Form changed:', event.detail);
  // Handle form changes
});

// Form rendered
document.addEventListener('formRender', (event) => {
  console.log('Form rendered:', event.detail.form);
  // Form is ready
});

Available Attributes

Attribute Type Description
form-id string Required. The ID of the form to display
api-key string Required. Your API key for authentication
submission-id string Optional. Load existing submission data
read-only boolean Optional. Display form in read-only mode
theme string Optional. Theme: 'bcgov', 'modern', or 'default'
language string Optional. Language code (e.g., 'en', 'fr')
base-url string Optional. Custom base URL for API calls

Framework Integration Examples

React

import React, { useEffect, useRef } from 'react';

function CHEFSForm({ formId, apiKey, onSuccess, onError }) {
  const formRef = useRef();

  useEffect(() => {
    const form = formRef.current;
    
    form.addEventListener('formSubmit', (event) => {
      onSuccess(event.detail.submission);
    });
    
    form.addEventListener('formError', (event) => {
      onError(event.detail.error);
    });
  }, [onSuccess, onError]);

  return (
    <chefs-form-viewer
      ref={formRef}
      form-id={formId}
      api-key={apiKey}
      theme="modern"
    />
  );
}

Vue

<template>
  <chefs-form-viewer
    ref="formRef"
    :form-id="formId"
    :api-key="apiKey"
    theme="bcgov"
    @formSubmit="handleSuccess"
    @formError="handleError"
  />
</template>

<script>
export default {
  data() {
    return {
      formId: 'abc123-def456',
      apiKey: 'your-api-key'
    };
  },
  methods: {
    handleSuccess(event) {
      this.$emit('submission-success', event.detail.submission);
    },
    handleError(event) {
      this.$emit('submission-error', event.detail.error);
    }
  }
};
</script>

WordPress

<?php
// In your WordPress theme or plugin
function add_chefs_form() {
    wp_enqueue_script('chefs-form-viewer', 
        'https://your-domain.com/widgets/chefs-form-viewer.js');
    
    return '<chefs-form-viewer 
              form-id="abc123-def456" 
              api-key="your-api-key"
              theme="bcgov">
            </chefs-form-viewer>';
}
add_shortcode('chefs_form', 'add_chefs_form');
?>
